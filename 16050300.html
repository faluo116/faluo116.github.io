简陋的.vimrc

set shiftwidth=4

colors desert 

syntax on

set tabstop=4

set cindent shiftwidth=4

set autoindent shiftwidth=4

set hlsearch

set nu

set encoding=utf-8

set fileencodings=utf-8,gbk,euc-jp

set backupskip=/tmp/*,/private/tmp/*

set laststatus=2

set nowrap



let mapleader="-"



nnoremap d dd

nnoremap ev :tab new $MYVIMRC
nnoremap sv :source $MYVIMRC


augroup ft_php

	autocmd!

	autocmd FileType php nnoremap  = I//
	autocmd FileType php :iabbrev  iff if(){}
	autocmd FileType php :iabbrev  forr for($i = 0 ; $i <  ; ++$i){}
augroup end



augroup ft_py

	autocmd!

	autocmd FileType python nnoremap  = I#
 	autocmd FileType python :iabbrev  pyhead #!/usr/bin/env python# -*- coding: utf-8 -*-
augroup end



set statusline=%=%F\ \%l/%L\ \%P\ -\
(Fin.)
